---
title: Trovo Users Dictionary
navigation.title: Trovo Users Dictionary
variables:
  - name: id
    type: string
    description: The user id of the user
    example: trovo_678296378
  - name: userName
    type: string
    description: The user name of the user
    example: trovouser123
  - name: display
    type: string
    description: The display name of the user
    example: TrovoUser123
  - name: role
    type: number
    description: The role of the user
    example: 1=`Viewer`, 2=`VIP`, 3=`Moderator`, 4=`Broadcaster`
  - name: isSubscribed
    type: boolean
    description: Is the user subscribed?
    example: `True` / `False`
  - name: isModerator
    type: boolean
    description: Is the user a moderator?
    example: `True` / `False`
---
